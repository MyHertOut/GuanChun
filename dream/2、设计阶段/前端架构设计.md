## 3. 核心架构设计

### 3.1 分层架构

```
┌─────────────────────────────────────┐
│         Views (页面视图)           │  ← 路由对应页面
│   Home.vue, Battle.vue, etc.      │
└──────────────┬────────────────────┘
               │
┌──────────────┼────────────────────┐
│              ▼                     │
│    Components (UI组件)           │  ← 可复用组件
│  业务组件 + 通用组件             │
└──────────────┬────────────────────┘
               │
┌──────────────┼────────────────────┐
│              ▼                     │
│   Composables (业务逻辑)         │  ← 逻辑复用
│  useBattle, usePet, etc.        │
└──────────────┬────────────────────┘
               │
┌──────────────┼────────────────────┐
│              ▼                     │
│      Stores (状态管理)           │  ← 全局状态
│  Pinia Store                   │
└──────────────┬────────────────────┘
               │
┌──────────────┼────────────────────┐
│              ▼                     │
│      API (数据层)                │  ← 接口调用
│  Axios封装                      │
└───────────────────────────────────┘
```

### 3.2 数据流向

```
用户交互
  ↓
View (组件)
  ↓
Composable (处理逻辑)
  ↓
Store (更新状态)
  ↓
API (请求数据)
  ↓
后端服务
  ↓
响应数据
  ↓
Store (更新状态)
  ↓
View (自动更新) ← 响应式
```

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { BattleState, BattleLog } from '@/types'

export const useBattleStore = defineStore('battle', () => {
  // 状态
  const state = ref<BattleState | null>(null)
  const logs = ref<BattleLog[]>([])
  const isAnimating = ref(false)

  // 计算属性
  const isPlayerTurn = computed(() => state.value?.isPlayerTurn ?? false)
  const battleStatus = computed(() => state.value?.battleStatus ?? 'idle')
  const round = computed(() => state.value?.round ?? 0)

  // 方法
  const setState = (newState: BattleState) => {
    state.value = newState
  }

  const addLog = (log: BattleLog) => {
    logs.value.push(log)
  }

  const reset = () => {
    state.value = null
    logs.value = []
    isAnimating.value = false
  }

  return {
    state,
    logs,
    isAnimating,
    isPlayerTurn,
    battleStatus,
    round,
    setState,
    addLog,
    reset
  }
})
```

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { PetInstance, PetData } from '@/types'

export const usePetStore = defineStore('pet', () => {
  // 状态
  const pets = ref<PetInstance[]>([])
  const activePet = ref<PetInstance | null>(null)
  const petDex = ref<PetData[]>([])
  const loading = ref(false)

  // 计算属性
  const petCount = computed(() => pets.value.length)
  const activePetStats = computed(() => activePet.value?.stats)

  // 方法
  const setPets = (newPets: PetInstance[]) => {
    pets.value = newPets
  }

  const setActivePet = (pet: PetInstance | null) => {
    activePet.value = pet
  }

  const updatePet = (petUid: string, updates: Partial<PetInstance>) => {
    const index = pets.value.findIndex(p => p.uid === petUid)
    if (index !== -1) {
      Object.assign(pets.value[index], updates)
    }
  }

  return {
    pets,
    activePet,
    petDex,
    loading,
    petCount,
    activePetStats,
    setPets,
    setActivePet,
    updatePet
  }
})
```

```typescript
import type { Player, InventoryItem } from '@/types'

export const usePlayerStore = defineStore('player', () => {
  // 状态
  const player = ref<Player | null>(null)
  const inventory = ref<InventoryItem[]>([])

  // 计算属性

  const soulPower = computed(() => player.value?.soulPower ?? 0)
  const maxSoulPower = computed(() => player.value?.maxSoulPower ?? 0)
  const gold = computed(() => player.value?.gold ?? 0)

    player.value = newPlayer
  }

    if (player.value) {
      player.value.soulPower = Math.max(0, player.value.soulPower + amount)
    }
  }

  const updateGold = (amount: number) => {
    if (player.value) {
      player.value.gold = Math.max(0, player.value.gold + amount)
    }
  }

  return {
    player,
    inventory,
    loading,
    isLoggedIn,
    soulPower,
    maxSoulPower,
    gold,
    setPlayer,
    updateSoulPower,
    updateGold
  }
```

### 5.5 UI Store (UI状态)

```typescript
// stores/ui.ts
import { defineStore } from 'pinia'
import { ref } from 'vue'

export const useUiStore = defineStore('ui', () => {
  // 全局加载状态
  const isLoading = ref(false)
  const loadingText = ref('')

  // Toast
  const toasts = ref<Array<{id: string, message: string, type: string}>>([])

  // Modal
  const modal = ref<{show: boolean, component?: any, props?: any}>({
    show: false
  })

  // 侧边栏
  const sidebarOpen = ref(false)

  const showLoading = (text: string = '加载中...') => {
    isLoading.value = true
    loadingText.value = text
  }

  const hideLoading = () => {
    isLoading.value = false
    loadingText.value = ''
  }

  const showToast = (message: string, type: 'success' | 'error' | 'warning' = 'success') => {
    const id = Date.now().toString()
    toasts.value.push({ id, message, type })
    setTimeout(() => {
      toasts.value = toasts.value.filter(t => t.id !== id)
    }, 3000)
  }

  const showModal = (component: any, props: any = {}) => {
    modal.value = { show: true, component, props }
  }

  const hideModal = () => {
    modal.value = { show: false }
  }

  const toggleSidebar = () => {
    sidebarOpen.value = !sidebarOpen.value
  }

  return {
    isLoading,
    loadingText,
    toasts,
    modal,
    sidebarOpen,
    showLoading,
    hideLoading,
    showToast,
    showModal,
    hideModal,
    toggleSidebar
  }
})
```

## 6. Composables 设计

### 6.1 useBattle (战斗逻辑)

```typescript
// composables/useBattle.ts
import { ref } from 'vue'
import { useBattleStore } from '@/stores/battle'
import { battleApi } from '@/api/battle'

export function useBattle() {
  const battleStore = useBattleStore()
  const loading = ref(false)
  const error = ref<Error | null>(null)

  // 开始战斗
  const startBattle = async (playerPetId: string, enemyPetId: string) => {
    loading.value = true
    error.value = null
    try {
      const result = await battleApi.start({ playerPetId, enemyPetId })
      battleStore.setState(result.state)
      battleStore.logs = result.logs
      return result
    } catch (e) {
      error.value = e as Error
      throw e
    } finally {
      loading.value = false
    }
  }

  // 提交行动
  const submitAction = async (action: BattleAction) => {
    loading.value = true
    try {
      const result = await battleApi.action(battleStore.state!.battleId, action)
      battleStore.setState(result.state)
      battleStore.logs.push(...result.logs)
      return result
    } finally {
      loading.value = false
    }
  }

  // 结束战斗
  const finishBattle = async () => {
    loading.value = true
    try {
      const result = await battleApi.finish(battleStore.state!.battleId)
      battleStore.reset()
      return result
    } finally {
      loading.value = false
    }
  }

  return {
    loading,
    error,
    startBattle,
    submitAction,
    finishBattle
  }
}
```

### 6.2 usePet (魂宠逻辑)

```typescript
// composables/usePet.ts
import { ref, onMounted } from 'vue'
import { usePetStore } from '@/stores/pet'
import { petApi } from '@/api/pet'

export function usePet(petId: string) {
  const petStore = usePetStore()
  const loading = ref(false)
  const error = ref<Error | null>(null)

  const fetchPet = async () => {
    loading.value = true
    error.value = null
    try {
      const pet = await petApi.getDetail(petId)
      return pet
    } catch (e) {
      error.value = e as Error
      throw e
    } finally {
      loading.value = false
    }
  }

  // 异变
  const evolve = async (evolutionId: string) => {
    loading.value = true
    try {
      const result = await petApi.evolve(petId, { evolutionId })
      petStore.updatePet(petId, result.newPet)
      return result
    } finally {
      loading.value = false
    }
  }

  // 升级
  const levelUp = async (exp: number) => {
    loading.value = true
    try {
      const result = await petApi.levelUp(petId, { exp })
      petStore.updatePet(petId, result.newPet)
      return result
    } finally {
      loading.value = false
    }
  }

  onMounted(() => {
    fetchPet()
  })

  return {
    loading,
    error,
    fetchPet,
    evolve,
    levelUp
  }
}
```

### 6.3 useStorage (本地存储)

```typescript
// composables/useStorage.ts
import { ref, watch } from 'vue'

export function useStorage<T>(key: string, defaultValue: T) {
  const storedValue = localStorage.getItem(key)
  const value = ref<T>(storedValue ? JSON.parse(storedValue) : defaultValue)

  watch(value, (newValue) => {
    localStorage.setItem(key, JSON.stringify(newValue))
  }, { deep: true })

  const remove = () => {
    localStorage.removeItem(key)
    value.value = defaultValue
  }

  return {
    value,
    remove
  }
}

// 使用示例
const { value: theme } = useStorage('theme', 'light')
```

## 7. API 封装设计

### 7.1 Axios 实例配置

```typescript
// api/index.ts
import axios, { AxiosError } from 'axios'
import { useUiStore } from '@/stores/ui'

// 创建 axios 实例
const api = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
api.interceptors.request.use(
  (config) => {
    // 添加 Token
    const token = localStorage.getItem('token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }

    // 显示加载
    const uiStore = useUiStore()
    if (config.showLoading !== false) {
      uiStore.showLoading()
    }

    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
api.interceptors.response.use(
  (response) => {
    const uiStore = useUiStore()
    uiStore.hideLoading()

    return response.data
  },
  (error: AxiosError) => {
    const uiStore = useUiStore()
    uiStore.hideLoading()

    // 处理错误
    if (error.response) {
      switch (error.response.status) {
        case 401:
          uiStore.showToast('未授权，请重新登录', 'error')
          // 跳转登录
          break
        case 403:
          uiStore.showToast('无权限访问', 'error')
          break
        case 404:
          uiStore.showToast('资源不存在', 'error')
          break
        case 500:
          uiStore.showToast('服务器错误', 'error')
          break
        default:
          uiStore.showToast('请求失败', 'error')
      }
    } else if (error.request) {
      uiStore.showToast('网络错误', 'error')
    } else {
      uiStore.showToast('请求配置错误', 'error')
    }

    return Promise.reject(error)
  }
)

export default api
```

### 7.2 战斗 API

```typescript
// api/battle.ts
import api from './index'
import type { BattleAction, BattleResult, BattleState } from '@/types'

export const battleApi = {
  // 开始战斗
  start: async (params: { playerPetId: string, enemyPetId: string }) => {
    return api.post<{ state: BattleState, logs: BattleLog[] }>('/battle/start', params)
  },

  // 提交行动
  action: async (battleId: string, action: BattleAction) => {
    return api.post<{ state: BattleState, logs: BattleLog[] }>(`/battle/${battleId}/action`, action)
```

```vue
  count?: number
```
